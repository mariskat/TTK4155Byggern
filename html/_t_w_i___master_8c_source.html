77 void TWI_Start_Transceiver_With_Data( unsigned char *msg, unsigned char msgSize )
78 {
79  unsigned char temp;
80 
81  while ( TWI_Transceiver_Busy()); // Wait until TWI is ready for next transmission.

84  TWI_buf[0] = msg[0]; // Store slave address with R/W setting.

88  TWI_buf[ temp ] = msg[ temp ];

90  TWI_statusReg.all = 0;
91  TWI_state = TWI_NO_STATE ;
92  TWCR = (1<<TWEN)| // TWI Interface enabled.
93  (1<<TWIE)|(1<<TWINT)| // Enable TWI Interupt and clear the flag.
94  (0<<TWEA)|(1<<TWSTA)|(0<<TWSTO)| // Initiate a START condition.
95  (0<<TWWC); //

114 /****************************************************************************
115 Call this function to read out the requested data from the TWI transceiver buffer. I.e. first call
116 TWI_Start_Transceiver to send a request for data to the slave. Then Run this function to collect the

139 /****************************************************************************
140 This function is the Interrupt Service Routine (ISR), and called when the TWI interrupt is triggered;
141 that is whenever a TWI event has occurred. This function should not be called directly from the main
142 application.
143 ****************************************************************************/
144 ISR(TWI_vect)
145 {
146  static unsigned char TWI_bufPtr;
147 
148  switch (TWSR)
149  {
150  case TWI_START: // START has been transmitted
151  case TWI_REP_START: // Repeated START has been transmitted
152  TWI_bufPtr = 0; // Set buffer pointer to the TWI Address location
153  case TWI_MTX_ADR_ACK: // SLA+W has been tramsmitted and ACK received
154  case TWI_MTX_DATA_ACK: // Data byte has been tramsmitted and ACK received
155  if (TWI_bufPtr < TWI_msgSize)
156  {
157  TWDR = TWI_buf[TWI_bufPtr++];
158  TWCR = (1<<TWEN)| // TWI Interface enabled
159  (1<<TWIE)|(1<<TWINT)| // Enable TWI Interupt and clear the flag to send byte
160  (0<<TWEA)|(0<<TWSTA)|(0<<TWSTO)| //
161  (0<<TWWC); //
162  }else // Send STOP after last byte
163  {
164  TWI_statusReg.lastTransOK = TRUE; // Set status bits to completed successfully.
165  TWCR = (1<<TWEN)| // TWI Interface enabled
166  (0<<TWIE)|(1<<TWINT)| // Disable TWI Interrupt and clear the flag
167  (0<<TWEA)|(0<<TWSTA)|(1<<TWSTO)| // Initiate a STOP condition.
168  (0<<TWWC); //
169  }
170  break;
171  case TWI_MRX_DATA_ACK: // Data byte has been received and ACK tramsmitted
172  TWI_buf[TWI_bufPtr++] = TWDR;
173  case TWI_MRX_ADR_ACK: // SLA+R has been tramsmitted and ACK received
174  if (TWI_bufPtr < (TWI_msgSize-1) ) // Detect the last byte to NACK it.
175  {
176  TWCR = (1<<TWEN)| // TWI Interface enabled
177  (1<<TWIE)|(1<<TWINT)| // Enable TWI Interupt and clear the flag to read next byte
178  (1<<TWEA)|(0<<TWSTA)|(0<<TWSTO)| // Send ACK after reception
179  (0<<TWWC); //
180  }else // Send NACK after next reception
181  {
182  TWCR = (1<<TWEN)| // TWI Interface enabled
183  (1<<TWIE)|(1<<TWINT)| // Enable TWI Interupt and clear the flag to read next byte
184  (0<<TWEA)|(0<<TWSTA)|(0<<TWSTO)| // Send NACK after reception
185  (0<<TWWC); //
186  }
187  break;
188  case TWI_MRX_DATA_NACK: // Data byte has been received and NACK tramsmitted
189  TWI_buf[TWI_bufPtr] = TWDR;
190  TWI_statusReg.lastTransOK = TRUE; // Set status bits to completed successfully.
191  TWCR = (1<<TWEN)| // TWI Interface enabled
192  (0<<TWIE)|(1<<TWINT)| // Disable TWI Interrupt and clear the flag
193  (0<<TWEA)|(0<<TWSTA)|(1<<TWSTO)| // Initiate a STOP condition.
194  (0<<TWWC); //

198  (1<<TWIE)|(1<<TWINT)| // Enable TWI Interupt and clear the flag
